# Requiremnt
 - node >= v10.15.3
 - npm >= v6.4.1
 - nodemon >= v1.19.1

 # Code convention
 - eslint standard

 # Set up

 - Clone project
 - cd project folder -> run "npm i" -> run "npm run postinstall"

 # Run Project
 - cd project folder -> run "node index :numberTimes", ex : "node index 2", the picture will be rotate 90 degrees 2 times

 # Test 
 - cd project folder -> run "npm test"
